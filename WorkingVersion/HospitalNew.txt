create schema if not exists "Hospital";

create table if not exists Clinic(Id SERIAL primary key,
                                  ClinicName varchar(255));

create table if not exists Ward(Id SERIAL primary key,
                                WardName varchar(255),
                                Capacity int,
                                ClinicId int,
                                constraint Fk_Ward_Clinic_ClinicId
                                foreign key (ClinicId) references Clinic(Id));

create table if not exists Address(Id SERIAL primary key,
                                   Street varchar(255),
                                   City varchar(255),
                                   ZIP int,
                                   Country varchar(255));
                               
create table if not exists Employee(Id SERIAL primary key,
                                    FirstName varchar(255),
                                    LastName varchar(255),
                                    DOB date,
                                    PhoneNumber int,
                                    Email varchar(255),
                                    AddressId int,
                                    WardId int,
                                    constraint Fk_Employee_Address_AddressId
                                    foreign key (AddressId) references Address(Id),
                                    constraint Fk_Employee_Ward_WardId
                                    foreign key (WardId) references Ward(Id));

create table if not exists Contract(Id SERIAL primary key,
                                    TypeOfContract varchar(255),
                                    WorkPosition varchar(255),
                                    ContractStart date,
                                    constraint Fk_Contract_Employee_EmployeeId
                                    foreign key (Id) references Employee(Id));
                                   
create table if not exists Patient(Id SERIAL primary key,
                                   FirstName varchar(255),
                                   LastName varchar(255),
                                   DOB date,
                                   PhoneNumber int,
                                   EmergencyContact int);

create table if not exists Appointment(Id SERIAL primary key,
                                       PatientId int,
                                       WardId int,
                                       AppointmentDate date,
                                       ApointmentTime time,
                                       constraint Fk_Appointment_Patient_PatientId
                                       foreign key (PatientId) references Patient(Id),
                                       constraint Fk_Appointment_Ward_WardId
                                       foreign key (WardId) references Ward(Id))
